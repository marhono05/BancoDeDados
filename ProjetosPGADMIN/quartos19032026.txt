create table cliente (
cod_cli serial,
nome_cliente varchar (40) not null,
rg_cliente char (10) not null,
cod_quarto int, 

constraint pk_cliente primary key (cod_cli)
);

create table quarto (
cod_quarto serial,
num_quarto int not null,
and_quarto int not null,
qnt_leito int not null, 

constraint pk_quarto primary key (cod_quarto)
);

create table clie_quar (
cod_cli Interger,
cod_quarto Interger


constraint pk_clie_quar primary key (cod_cli, cod_quarto),
constraint fk_cliente_cliequar foreign key (cod_cli) references cliente (cod_cli),
constraint fk_quarto_cliequar foreign key (cod_quarto) references quarto (cod_quarto)
);
