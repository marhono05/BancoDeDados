create table departamento (
id_depa serial,
desc_depa varchar (30) not null,
tipo_depa varchar (10) not null,

constraint pk_departamento primary key (id_depa)
);

create table funcionario (
id_func serial,
nome_func char (40) not null,
cid_func varchar (20) not null,
id_depa int,
id_escr int, 

constraint pk_funcionario primary key (id_func),

constraint fk_depa_func foreign key (id_depa) references departamento (id_depa),
constraint fk_escr_func foreign key (id_escr) references escritorio (id_escr)
);

create table escritorio (
id_escr serial,
nome_escr char (30) not null,
end_escri char (50) not null,

constraint pk_escritorio primary key (id_escr)
);

/* Quando for relacionamento N para N , necessário criar outra tabela*/
create table depa_escr (
id_escr Interger,
id_depa Interger 

constraint pk_depa_escr primary key (id_escr, id_depa),
constraint fk_departamento_depaescr foreign key (id_depa) references departamento (id_depa),
constraint fk_escritorio_depaescr foreign key (id_escr) references escritorio (id_escr)
);
